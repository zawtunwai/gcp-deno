async function sendMessage(chatId, text, parseMode = null) {
  const url = `https://api.telegram.org/bot${BOT_TOKEN}/sendMessage`;
  const body = {
    chat_id: chatId,
    text: text,
    disable_web_page_preview: true
  };
  if (parseMode) body.parse_mode = parseMode;
  
  const response = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify(body)
  });
  return response.json();
}

async function sendPhoto(chatId, photoUrl, caption = "") {
  const url = `https://api.telegram.org/bot${BOT_TOKEN}/sendPhoto`;
  const response = await fetch(url, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({
      chat_id: chatId,
      photo: photoUrl,
      caption: caption
    })
  });
  return response.json();
}

function isValidUUID(uuid) {
  const uuidRegex = /^[0-9a-f]{8}-[0-9a-f]{4}-[4][0-9a-f]{3}-[89ab][0-9a-f]{3}-[0-9a-f]{12}$/i;
  return uuidRegex.test(uuid);
}

function generateUUID() {
  return crypto.randomUUID();
}

function formatVLessConfig(uuid, host, credit = "GCP-Deno") {
  return `vless://${uuid}@${host}:443?encryption=none&security=tls&sni=${host}&fp=randomized&type=ws&host=${host}&path=%2F%3Fed%3D2048#${credit}`;
}

function formatClashConfig(uuid, host) {
  return `- type: vless
  name: ${host}
  server: ${host}
  port: 443
  uuid: ${uuid}
  network: ws
  tls: true
  udp: false
  sni: ${host}
  client-fingerprint: chrome
  ws-opts:
    path: "/?ed=2048"
    headers:
      host: ${host}`;
}

        "utls": {
          "enabled": true,

        "path": "/?ed=2048",

// ===== User Management Functions (KV သုံးမယ်) =====
async function isAuthorized(userId, kv) {
  if (userId === OWNER_ID) return true;
  const allowed = await kv.get(`allowed:${userId}`);
  return allowed.value === true;
}

async function addAllowedUser(userId, kv) {
  await kv.put(`allowed:${userId}`, true);
}

async function removeAllowedUser(userId, kv) {
  await kv.put(`allowed:${userId}`, false);
}

async function getAllowedUsers(kv) {
  const list = await kv.list({ prefix: "allowed:" });
  const users = [];
  for await (const key of list) {
    const userId = key.name.replace("allowed:", "");
    if (userId && !isNaN(parseInt(userId))) {
      users.push(parseInt(userId));
    }
  }
  return users;
}

// ===== Config Management Functions =====
async function createConfig(userId, name, limitGB, expiryDays, kv) {
  const uuid = generateUUID();
  const expiry = Date.now() + (expiryDays * 24 * 60 * 60 * 1000);
  const limitBytes = limitGB * 1024 * 1024 * 1024;
  
  const config = {
    uuid: uuid,
    name: name,
    ownerId: userId,
    limitGB: limitGB,
    limitBytes: limitBytes,
    expiry: expiry,
    usageBytes: 0,
    createdAt: Date.now(),
    isActive: true
  };
  
  await kv.put(`config:${uuid}`, JSON.stringify(config));
  await kv.put(`userConfigs:${userId}:${uuid}`, true);
  
  return config;
}

async function getUserConfigs(userId, kv) {
  const configs = [];
  const list = await kv.list({ prefix: `userConfigs:${userId}:` });
  for await (const key of list) {
    const uuid = key.name.split(":").pop();
    const configData = await kv.get(`config:${uuid}`);
    if (configData.value) {
      const config = JSON.parse(configData.value);
      configs.push(config);
    }
  }
  return configs;
}

async function getConfigByUuid(uuid, kv) {
  const data = await kv.get(`config:${uuid}`);
  if (data.value) {
    return JSON.parse(data.value);
  }
  return null;
}

async function deleteConfig(uuid, kv) {
  const config = await getConfigByUuid(uuid, kv);
  if (config) {
    await kv.delete(`config:${uuid}`);
    await kv.delete(`userConfigs:${config.ownerId}:${uuid}`);
    return true;
  }
  return false;
}

async function updateConfigUsage(uuid, bytes, kv) {
  const configData = await kv.get(`config:${uuid}`);
  if (configData.value) {
    const config = JSON.parse(configData.value);
    config.usageBytes += bytes;
    await kv.put(`config:${uuid}`, JSON.stringify(config));
    return config;
  }
  return null;
}

function formatConfigMessage(config, host) {
  const remainingGB = ((config.limitBytes - config.usageBytes) / (1024 * 1024 * 1024)).toFixed(2);
  const expiryDate = new Date(config.expiry).toLocaleDateString("en-GB");
  const status = config.isActive && config.expiry > Date.now() && config.usageBytes < config.limitBytes ? "✅ Active" : "❌ Expired/Limit Reached";
  
  const vlessUri = formatVLessConfig(config.uuid, host, "GCP-Deno");
  const clashConfig = formatClashConfig(config.uuid, host);
  const singboxConfig = formatSingboxConfig(config.uuid, host);
  
  return `📁 *Config: ${config.name}*
🆔 *UUID:* \`${config.uuid}\`
📊 *Status:* ${status}
💾 *Usage:* ${(config.usageBytes / (1024 * 1024 * 1024)).toFixed(2)} GB / ${config.limitGB} GB
📉 *Remaining:* ${remainingGB} GB
⏰ *Expiry:* ${expiryDate}

🔗 *VLESS URI:*
\`${vlessUri}\`

📡 *Clash-Meta:*
\`\`\`yaml
${clashConfig}
\`\`\`

🎮 *Sing-box:*
\`\`\`json
${singboxConfig}
\`\`\`

🔗 *Config URL:* https://${host}/${config.uuid}`;
}

async function handleNewConfig(chatId, args, userId, kv) {
  if (args.length < 3) {
    await sendMessage(chatId, "❌ Usage: /new <name> <limitGB> <days>\n\nExample: /new myserver 100 30");
    return;
  }
  
  const name = args[0];
  const limitGB = parseInt(args[1]);
  const days = parseInt(args[2]);
  
  if (isNaN(limitGB) || limitGB < 1 || limitGB > 10000) {
    await sendMessage(chatId, "❌ Limit must be between 1-10000 GB");
    return;
  }
  
  if (isNaN(days) || days < 1 || days > 365) {
    await sendMessage(chatId, "❌ Days must be between 1-365");
    return;
  }
  
  const config = await createConfig(userId, name, limitGB, days, kv);
  const message = formatConfigMessage(config, DENO_DOMAIN);
  await sendMessage(chatId, message, "Markdown");
}

async function handleListConfigs(chatId, userId, kv) {
  const configs = await getUserConfigs(userId, kv);
  
  if (configs.length === 0) {
    await sendMessage(chatId, "📭 You don't have any configs yet.\n\nUse /new to create one!");
    return;
  }
  
  let message = "*📋 Your Configs:*\n\n";
  for (const config of configs) {
    const status = config.isActive && config.expiry > Date.now() && config.usageBytes < config.limitBytes ? "✅" : "❌";
    message += `${status} *${config.name}*\n🆔 \`${config.uuid}\`\n📊 ${(config.usageBytes / (1024 * 1024 * 1024)).toFixed(2)}GB/${config.limitGB}GB\n\n`;
  }
  
  await sendMessage(chatId, message, "Markdown");
}

async function handleInfo(chatId, args, userId, kv) {
  if (args.length < 1) {
    await sendMessage(chatId, "❌ Usage: /info <uuid>");
    return;
  }
  
  const uuid = args[0];
  if (!isValidUUID(uuid)) {
    await sendMessage(chatId, "❌ Invalid UUID format");
    return;
  }
  
  const config = await getConfigByUuid(uuid, kv);
  if (!config) {
    await sendMessage(chatId, "❌ Config not found");
    return;
  }
  
  if (config.ownerId !== userId && userId !== OWNER_ID) {
    await sendMessage(chatId, "❌ You don't own this config");
    return;
  }
  
  const message = formatConfigMessage(config, DENO_DOMAIN);
  await sendMessage(chatId, message, "Markdown");
}

async function handleListUsers(chatId, userId, kv) {

  }
  
  const users = await getAllowedUsers(kv);

async function handleBroadcast(chatId, args, userId, kv) {
  if (userId !== OWNER_ID) {
    await sendMessage(chatId, "❌ Only owner can use this command");
    return;
  }
  
  if (args.length < 1) {
    await sendMessage(chatId, "❌ Usage: /broadcast <message>");
    return;
  }
  
  const message = args.join(" ");
  const users = await getAllowedUsers(kv);
  users.push(OWNER_ID);
  
  let success = 0;
  let failed = 0;
  
  for (const uid of users) {
    try {
      await sendMessage(uid, `📢 *Announcement:*\n\n${message}`, "Markdown");
      success++;
    } catch (e) {
      failed++;
    }
    await new Promise(r => setTimeout(r, 50)); // rate limit
  }
  
  await sendMessage(chatId, `✅ Broadcast sent!\n📨 Success: ${success}\n❌ Failed: ${failed}`);
}

// ===== Main Handler =====
async function handleUpdate(update, kv) {
  if (!update.message) return;
  
  const message = update.message;
  const chatId = message.chat.id;
  const userId = message.from.id;
  const text = message.text || "";
  
  // Check authorization (except /start)
  if (text !== "/start") {
    const authorized = await isAuthorized(userId, kv);
    if (!authorized) {
      await sendMessage(chatId, "❌ You are not authorized to use this bot.\nContact the owner for access.");
      return;
    }
  }
  
  const parts = text.split(" ");
  const command = parts[0].toLowerCase();
  const args = parts.slice(1);
  
  switch (command) {
    case "/start":
      await handleStart(chatId, kv);
      break;
    case "/new":
      await handleNewConfig(chatId, args, userId, kv);
      break;
    case "/list":
      await handleListConfigs(chatId, userId, kv);
      break;
    case "/info":
      await handleInfo(chatId, args, userId, kv);
      break;
    case "/delete":
      await handleDelete(chatId, args, userId, kv);
      break;
    case "/adduser":
      await handleAddUser(chatId, args, userId, kv);
      break;
    case "/removeuser":
      await handleRemoveUser(chatId, args, userId, kv);
      break;
    case "/users":
      await handleListUsers(chatId, userId, kv);
      break;
    case "/broadcast":
      await handleBroadcast(chatId, args, userId, kv);
      break;
    default:
      await sendMessage(chatId, "❌ Unknown command. Use /start to see available commands.");
  }
}

// ===== Cloudflare Worker Entry Point =====
export default {
  async fetch(request, env) {
    const url = new URL(request.url);
    
    // Webhook endpoint
    if (url.pathname === "/webhook" && request.method === "POST") {
      const update = await request.json();
      await handleUpdate(update, env.GCP_DENO_KV);
      return new Response("OK", { status: 200 });
    }
    
    // Set webhook endpoint
    if (url.pathname === "/setwebhook") {
      const webhookUrl = `https://${url.hostname}/webhook`;
      const response = await fetch(`https://api.telegram.org/bot${BOT_TOKEN}/setWebhook?url=${webhookUrl}`);
      const result = await response.json();
      return new Response(JSON.stringify(result), {
        headers: { "Content-Type": "application/json" }
      });
    }
    
    // Health check
    if (url.pathname === "/") {
      return new Response("GCP Deno Bot is running! Created by မောင်သုည", { status: 200 });
    }
    
    return new Response("Not found", { status: 404 });
  }
};
